const MAIN_FOLDER_ID = "1c2M4eaAXJP9BtY966Y4DP3cVWnQ3SxkV"; const SHEET_ID = "1niGviqWvuLgEja1laZgJ3ghWgvswFNiu0ynHlHGV2HQ"; const MAX_UPLOADS = 3; const MAX_VIEWERS = 3; const SESSION_TIMEOUT = 60000; // 60 seconds /* =========================== UPLOAD API (POST) =========================== */ function doPost(e) { try { const data = JSON.parse(e.postData.contents); const productId = data.productId; const base64Data = data.file; const fileName = data.fileName; if (!productId) { throw new Error("Product ID missing"); } const parentFolder = DriveApp.getFolderById(MAIN_FOLDER_ID); const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet(); const sheetData = sheet.getDataRange().getValues(); let rowIndex = -1; let uploadCount = 0; let oldFileId = null; // Check if ProductID exists for (let i = 1; i < sheetData.length; i++) { if (sheetData[i][0] === productId) { rowIndex = i + 1; uploadCount = Number(sheetData[i][3]) || 0; oldFileId = sheetData[i][1]; break; } } // Block if max uploads reached if (uploadCount >= MAX_UPLOADS) { return ContentService.createTextOutput( JSON.stringify({ status: "blocked", message: "Upload limit reached (3 lifetime uploads used)" }) ).setMimeType(ContentService.MimeType.JSON); } // Delete old file if (oldFileId) { try { DriveApp.getFileById(oldFileId).setTrashed(true); } catch (err) { Logger.log("Old file deletion failed: " + err); } } // Get or create product folder let folders = parentFolder.getFoldersByName(productId); let productFolder = folders.hasNext() ? folders.next() : parentFolder.createFolder(productId); // Decode video const decoded = Utilities.base64Decode(base64Data); const blob = Utilities.newBlob(decoded, "video/mp4", fileName); const file = productFolder.createFile(blob); // Make file public file.setSharing( DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW ); const fileId = file.getId(); const videoURL = "https://drive.google.com/uc?export=download&id=" + fileId; const newUploadCount = uploadCount + 1; // Update existing row OR create new row if (rowIndex > -1) { sheet.getRange(rowIndex, 2, 1, 4).setValues([ [fileId, videoURL, newUploadCount, new Date()] ]); } else { sheet.appendRow([ productId, fileId, videoURL, 1, new Date(), "" // session column ]); } return ContentService.createTextOutput( JSON.stringify({ status: "success", uploadCount: newUploadCount, remaining: MAX_UPLOADS - newUploadCount }) ).setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput( JSON.stringify({ status: "error", message: err.toString() }) ).setMimeType(ContentService.MimeType.JSON); } } /* =========================== FETCH VIDEO API (GET) WITH 3 DEVICE LIMIT =========================== */ function doGet(e) { try { const productId = e.parameter.id; const deviceId = e.parameter.device; if (!productId || !deviceId) { throw new Error("Missing parameters"); } const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet(); const data = sheet.getDataRange().getValues(); const now = new Date().getTime(); for (let i = 1; i < data.length; i++) { if (data[i][0] === productId) { let sessions = []; if (data[i][5]) { sessions = JSON.parse(data[i][5]); } // Remove expired sessions sessions = sessions.filter(s => { return now - s.time < SESSION_TIMEOUT; }); // Check if device already exists let exists = sessions.find(s => s.device === deviceId); if (!exists) { if (sessions.length >= MAX_VIEWERS) { return ContentService.createTextOutput( JSON.stringify({ status: "blocked", message: "Maximum 3 viewers already watching" }) ).setMimeType(ContentService.MimeType.JSON); } sessions.push({ device: deviceId, time: now }); } // Save updated sessions sheet.getRange(i + 1, 6).setValue(JSON.stringify(sessions)); return ContentService.createTextOutput( JSON.stringify({ status: "allowed", videoURL: data[i][2], viewers: sessions.length }) ).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput( JSON.stringify({ status: "not_found" }) ).setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput( JSON.stringify({ status: "error", message: err.toString() }) ).setMimeType(ContentService.MimeType.JSON); } }
